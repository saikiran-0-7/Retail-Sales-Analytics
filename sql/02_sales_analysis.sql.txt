-- Total Sales
SELECT ROUND(SUM(Sales),2) AS Total_Sales
FROM superstore1;

-- Total Profit
SELECT ROUND(SUM(Profit),2) AS Total_Profit
FROM superstore1;

-- Total Orders
SELECT COUNT(DISTINCT `Order ID`) AS Total_Orders
FROM superstore1;

-- Average Order Value
SELECT ROUND(
SUM(Sales)/
COUNT(DISTINCT `Order ID`)
,2) AS Average_Order_Value
FROM superstore1;

-- Average Profit
SELECT ROUND(AVG(Profit),2)
AS Average_Profit
FROM superstore1;

-- Total Quantity Sold
SELECT SUM(Quantity)
AS Total_Quantity
FROM superstore1;

-- Total Discount
SELECT ROUND(SUM(Discount),2)
AS Total_Discount
FROM superstore1;
